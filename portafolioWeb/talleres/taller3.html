2. Realizar la siguiente figura por medio del ciclo Mientras – Hacer

         *
        **
       ***
      ****
     *****
    ******
   *******
  ********
 *********
********** 


        Proceso sin_titulo
            Definir control Como Logico;
            Definir incremento Como Entero;
            Definir rama Como Caracter;
            Definir espacios Como Caracter;
            Definir numeroEspacios Como Entero;
            Definir meta Como Entero;
            
            rama <- "*";
            control <- Verdadero;
            incremento <- 1;
            numeroEspacios <- 10;
            
            Mientras control = Verdadero Hacer
                
                meta <- 1;
                espacios <- " ";
                Mientras meta < numeroEspacios Hacer
                    espacios <- Concatenar(" " , espacios);
                    meta <- meta + 1;
                FinMientras
                
                
                Escribir espacios , rama;
                Si incremento >= 10 Entonces
                    control <- falso;
                FinSi
                
                rama <- Concatenar(rama,'*');
                incremento <- incremento + 1;
                numeroEspacios <- numeroEspacios - 1;
            FinMientras
        FinProceso

            control <- Verdadero;
            incremento <- 1;
            numeroEspacios <- 9;
            rama <- "*";
            
            Escribir "          ",rama;
            Repetir
                
                meta <- 1;
                espacios <- " ";
                Mientras meta < numeroEspacios Hacer
                    espacios <- Concatenar(" " , espacios);
                    meta <- meta + 1;
                FinMientras
                
                rama <- Concatenar("*", rama);
                rama <- Concatenar(rama, "*");

                incremento <- incremento + 1;
                Si incremento >= 10 Entonces
                    control <- Falso;

            Escribir("         ***");
            Escribir("        *****");
